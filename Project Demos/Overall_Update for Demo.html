# Separate features (X) and target (y)
X = df.drop(columns=['Class'])
y = df['Class']

# Remove Constant Features
vt = VarianceThreshold(0.03)  # 0.1-0.01 You can specify a threshold if desired
vt.fit(X)
X_filtered = X.loc[:, vt.get_support()]

# Identify removed features
removed_features = X.columns[~vt.get_support()]
print("Number of features removed by variance thresholding:", len(removed_features))
print("Features removed by variance thresholding:", removed_features)

# Print the number of selected features after variance thresholding
print("Number of features after variance thresholding:", X_filtered.shape[1])

## Specifies the number of top features (genes) to select using SelectKBest
## Changes the complexity of the model. A smaller number might lead to underfitting, while a larger number could lead to overfitting.
# Feature selection using SelectKBest
num_features = 10  # can Change value
selector = SelectKBest(score_func=f_classif, k=num_features)
selector.fit(X_filtered, y)

# Get the selected feature names
selected_features = X_filtered.columns[selector.get_support()]

# Print the selected features
print("Selected features:", selected_features)

# Visualize feature importances
plt.figure(figsize=(10, 6))
sns.barplot(x=selector.scores_[selector.get_support()], y=selected_features)
plt.title('Feature Importances (SelectKBest)')
plt.xlabel('Importance Score')
plt.ylabel('Feature')
plt.show()

# Transform the features
X_selected = selector.transform(X_filtered)
