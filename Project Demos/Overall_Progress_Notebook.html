def train_model(model, X_train, y_train, X_test, y_test, model_name):
    """Trains and evaluates a model and returns metrics and feature importances."""
    model.fit(X_train, y_train)
    y_pred = model.predict(X_test)

    print(classification_report(y_test, y_pred))
    balanced_accuracy = balanced_accuracy_score(y_test, y_pred)
    print("Balanced Accuracy:", balanced_accuracy)

    # Confusion Matrix Visualization
    cm = confusion_matrix(y_test, y_pred)
    plt.figure(figsize=(6, 4))
    sns.heatmap(cm, annot=True, fmt="d", cmap="Blues")
    plt.title(f'{model_name} Confusion Matrix')
    plt.xlabel('Predicted Label')
    plt.ylabel('True Label')
    plt.show()

    # REMOVE FEATURE IMPORTANCE CODE FOR SVM
    if model_name == "SVM":
        return model, balanced_accuracy  # Return early for SVM

    # Feature Importance (using permutation importance for SVM, built-in for others)
    feature_names = df.drop(columns=['Class', 'binary_class']).columns # Get feature names
    if hasattr(model, 'feature_importances_'):
        importances = model.feature_importances_
    else:
        print("Feature importances not available for this model.")
        return model, balanced_accuracy

    # Plot feature importances
    feature_importances = sorted(zip(feature_names, importances), key=lambda x: x[1], reverse=True)

    print(f"\nFeature Importances ({model_name}):")
    for feature, importance in feature_importances[:10]:  # Print top 10
        print(f"{feature}: {importance}")

    num = min(len(feature_importances), 20)  # Number of features to display
    plt.figure(figsize=(10, 5))
    plt.title(f"{model_name} Feature importances")
    plt.bar(range(num), [x[1] for x in feature_importances[:num]])
    plt.xticks(range(num), [x[0] for x in feature_importances[:num]], rotation=90)
    plt.xlim([-1, num])
    plt.ylabel("Importance")
    plt.xlabel("Feature")
    plt.show()

    return model, balanced_accuracy
