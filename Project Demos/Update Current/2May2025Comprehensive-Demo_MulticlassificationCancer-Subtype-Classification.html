# One-vs-Rest ROC AUC score
rf_probs = rf_model.predict_proba(X_test_scaled)
macro_roc_auc_ovo = roc_auc_score(y_test, rf_probs, multi_class="ovo", average="macro")
print(f"One-vs-Rest ROC AUC score:\n{macro_roc_auc_ovo}")

# Binarize labels for multiclass ROC
y_test_bin = label_binarize(y_test, classes=[0, 1, 2, 3, 4])
n_classes = y_test_bin.shape[1]
fpr, tpr, roc_auc = {}, {}, {}

for i in range(n_classes):
    fpr[i], tpr[i], _ = roc_curve(y_test_bin[:, i], rf_probs[:, i])
    roc_auc[i] = auc(fpr[i], tpr[i])

# Micro/macro-average ROC
fpr["micro"], tpr["micro"], _ = roc_curve(y_test_bin.ravel(), rf_probs.ravel())
roc_auc["micro"] = auc(fpr["micro"], tpr["micro"])
all_fpr = np.unique(np.concatenate([fpr[i] for i in range(n_classes)]))
mean_tpr = np.zeros_like(all_fpr)
for i in range(n_classes):
    mean_tpr += np.interp(all_fpr, fpr[i], tpr[i])
mean_tpr /= n_classes
fpr["macro"] = all_fpr
tpr["macro"] = mean_tpr
roc_auc["macro"] = auc(fpr["macro"], tpr["macro"])

# Plot ROC curves
plt.figure(figsize=(10,8))
plt.plot(fpr["micro"], tpr["micro"],
         label=f"micro-average ROC curve (area = {roc_auc['micro']:.2f})",
         color="deeppink", linestyle=":", linewidth=4)
plt.plot(fpr["macro"], tpr["macro"],
         label=f"macro-average ROC curve (area = {roc_auc['macro']:.2f})",
         color="navy", linestyle=":", linewidth=4)
colors = plt.cm.viridis(np.linspace(0, 1, n_classes))
for i, color in zip(range(n_classes), colors):
    plt.plot(fpr[i], tpr[i], color=color, lw=2,
             label=f"ROC curve of class {cancer_names[i]} (area = {roc_auc[i]:.2f})")
plt.plot([0, 1], [0, 1], "k--", lw=2)
plt.xlim([0.0, 1.0])
plt.ylim([0.0, 1.05])
plt.xlabel("False Positive Rate")
plt.ylabel("True Positive Rate")
plt.title("Receiver Operating Characteristic (ROC) Curves")
plt.legend(loc="lower right")
plt.show()

# ROC AUC stands for Receiver Operating Characteristic - Area Under the Curve. It measures how well your model distinguishes between classes. # In multiclass problems, the "one-vs-rest" (OvR) approach computes the ROC AUC for each class vs. all other classes, then averages these scores # (macro-average). Interpretation: 1.0 = Perfect class separation (your model never confuses any class). 0.5 = Model is no better than random guessing. # > 0.9 = Excellent performance. Random Forest model perfectly distinguishes each cancer subtype from the others on the test set. # ROC AUC is threshold-independent: It evaluates model quality across all possible classification thresholds. # Macro-averaged OvR ROC AUC is a robust, standard metric for multiclass classification performance.# --- 10.
