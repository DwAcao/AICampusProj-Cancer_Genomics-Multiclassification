# --- 3. Data Partitioning ---
# Separate features and target.
X = df.drop('Class', axis=1)
y = df['Class']

# Split data into training and testing sets (80/20 split).
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.2, stratify=y, random_state=42
)
In[90]:
# --- 4. Feature Selection ---
# Remove zero-variance features.
vt = VarianceThreshold(threshold=0.3)
X_train_vt = vt.fit_transform(X_train)
X_test_vt = vt.transform(X_test)
In[91]:
# Select top features using SelectKBest.
kbest = SelectKBest(f_classif, k=650)
X_train_selected = kbest.fit_transform(X_train_vt, y_train)
X_test_selected = kbest.transform(X_test_vt)

X_test_scaled = scaler.transform(X_test_selected)

# --- 6. Model Training and Evaluation ---
# --- 6.1: Random Forest ---
# Train Random Forest model with GridSearchCV.
cv = StratifiedKFold(n_splits=3, shuffle=True, random_state=42)
param_grid = {
    'n_estimators': [100, 200, 300],
    'max_depth': [10, 15, 20],
    'min_samples_split': [2, 4],
    'min_samples_leaf': [1, 2]
}
grid_search = GridSearchCV(
    estimator=RandomForestClassifier(random_state=42),
    param_grid=param_grid,
    scoring='balanced_accuracy',
    cv=cv,
    n_jobs=-1
)
grid_search.fit(X_train_scaled, y_train)
best_params = grid_search.best_params_
best_rf = grid_search.best_estimator_
print("Random Forest Best Parameters:", best_params)

Random Forest Best Parameters: {'max_depth': 10, 'min_samples_leaf': 1, 'min_samples_split': 2, 'n_estimators': 100}

features_after_vt = X.columns[vt_mask]  # Original feature names filtered by VarianceThreshold
