# 2. Data Cleaning and Preprocessing
# Separate features (X) and target (y)
X = df.drop(columns=['Class'])
y = df['Class']

# 3. Impute missing values (if any) with the mean
imputer = SimpleImputer(strategy='mean')
X = imputer.fit_transform(X)

# Convert to DataFrame (for feature names)
X = pd.DataFrame(X, columns=df_x.columns)

# 4. Remove Constant Features
constant_filter = VarianceThreshold(threshold=0)
constant_filter.fit(X)
X_filtered = X.loc[:, constant_filter.get_support()]
print("Number of features after removing constant features:", X_filtered.shape[1])

# 5. Feature Selection (SelectKBest with f_classif)
selector = SelectKBest(f_classif, k=500)  # Specify the number of features to select
selector.fit(X_filtered, y)
X_selected = selector.transform(X_filtered)  # Transform AFTER fitting

# 5.1 Get the selected feature names
selected_features = X_filtered.columns[selector.get_support()]

print("Number of selected features:", X_selected.shape[1])
print("Selected features:", selected_features)

# 6. Data Scaling
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X_selected)

# 7. Split Data
X_train, X_test, y_train, y_test = train_test_split(
    X_scaled, y, test_size=0.2, stratify=y, random_state=20
)
